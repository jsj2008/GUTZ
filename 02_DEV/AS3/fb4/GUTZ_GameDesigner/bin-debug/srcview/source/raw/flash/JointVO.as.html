package  {
    
    import flash.geom.Point;
    
    public class JointVO {
        
        public var id:int;
        public var bodyA:int;
        public var bodyB:int;
        public var offsetA_pt:Point;
        public var offsetB_pt:Point;
        public var str:Number;
        public var dmp:Number;
        
        public function JointVO(i:int, body1:int, body2:int, offset1:Point, offset2:Point, s:Number, d:Number) {
            id = i;
            bodyA = body1;
            bodyB = body2;
            offsetA_pt = new Point(offset1.x, offset1.y);
            offsetB_pt = new Point(offset2.x, offset2.y);
            str = s;
            dmp = d;
        }
    }
}
